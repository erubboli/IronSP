      def on_case(arg, *cases)
        result = [:multi, [:code, arg ? "case (#{arg})" : 'case']]
        cases.map do |c|
          condition, exp = c
          result << [:code, condition == :else ? 'else' : "when #{condition}"] << compile(exp)
        end
        result << [:code, 'end']
        result
      end
