      def on_multi(*exps)
        case options[:trim_mode]
        when '>'
          i = 0
          while i < exps.size
            exps.delete_at(i + 1) if code?(exps[i]) && exps[i+1] == [:static, "\n"]
            i += 1
          end
        when '<>'
          i = 0
          while i < exps.size
            exps.delete_at(i + 1) if code?(exps[i]) && exps[i+1] == [:static, "\n"] &&
                                     (!exps[i-1] || (exps[i-1] == [:newline]))
            i += 1
          end
        end
        [:multi, *exps]
      end
