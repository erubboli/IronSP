    def CreateChildControls
      scope = Microsoft::SharePoint::Utilities::SPMonitoredScope.new "CreateChildControls #{self.class.name}"
      unless self.page.nil?
        ctrl = self.page.parse_control(to_html)
        self.controls.add(ctrl)
      end
    rescue Exception => ex
      logger.error ex
    ensure
      scope.dispose
    end
