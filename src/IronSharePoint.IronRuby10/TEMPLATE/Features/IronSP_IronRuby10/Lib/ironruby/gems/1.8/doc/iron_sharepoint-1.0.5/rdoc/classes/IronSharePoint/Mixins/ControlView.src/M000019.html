    def error_template ex
      backtrace = ex.backtrace[0..2].map{|x| "<div>#{x}</div>"}
      "<div class=\"ironsp-error\" style=\"display: inline-block; color: red; font-size: 10px;line-height: 110%;text-align: left;\">\n<p>\#{ex.message}</p>\n\#{backtrace}\n</div>\n"
    end
