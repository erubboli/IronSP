      def on_dynamic(code)
        if @pretty
          @last = :noindent
          tmp = unique_name
          gsub_code = if ''.respond_to?(:html_safe?)
                        "#{tmp} = #{tmp}.html_safe? ? #{tmp}.gsub(\"\\n\", #{indent.inspect}).html_safe : #{tmp}.gsub(\"\\n\", #{indent.inspect})"
                      else
                        "#{tmp} = #{tmp}.gsub(\"\\n\", #{indent.inspect})"
                      end
          [:multi,
           [:code, "#{tmp} = (#{code}).to_s"],
           [:code, "if #{@pre_tags_name} !~ #{tmp}; #{gsub_code}; end"],
           [:dynamic, tmp]]
        else
          [:dynamic, code]
        end
      end
