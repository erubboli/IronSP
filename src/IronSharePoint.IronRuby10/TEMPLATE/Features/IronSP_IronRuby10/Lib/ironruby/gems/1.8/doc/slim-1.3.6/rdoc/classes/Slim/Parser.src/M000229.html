    def deprecated_shortcut(v)
      warn "Slim :shortcut string values are deprecated, use hash like { '#' => { :tag => 'div', :attr => 'id' } }"
      if v =~ /\A([^\s]+)\s+([^\s]+)\Z/
        { :tag => $1, :attr => $2 }
      else
        { :attr => v }
      end
    end
